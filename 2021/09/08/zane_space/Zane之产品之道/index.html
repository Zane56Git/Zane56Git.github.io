电商平台用户行为数据分析（MySQL）

https://zhuanlan.zhihu.com/p/356326256

CREATE TABLE userbehavior_dwd(
user_id VARCHAR(255) COMMENT ‘用户id’ ,
item_id VARCHAR(255) COMMENT ‘商品id’ ,
category_id VARCHAR(255) COMMENT ‘商品类型id’ ,
behavior_type VARCHAR(255) COMMENT ‘访问类型：pv,cart,fav,buy（浏览，添加购物车，收藏，购买）’ ,
timestamps DATETIME COMMENT ‘行为发生时间戳’ ,
behavior_date VARCHAR(255) COMMENT ‘日期’ ,
behavior_time VARCHAR(255) COMMENT ‘时间’ ,
weekdays VARCHAR(255) COMMENT ‘周-天数’
) COMMENT = ‘用户行为表(维度表)’;

select * from userbehavior_dwd

select min(behavior_date),max(behavior_date) from userbehavior_dwd

delete from userbehavior_dwd where behavior_date>’2017-11-25’

– ▶▶用户行为指标
– 一、按实际维度
– 1、用户活跃度（这里就涉及每天、每时，不会是全量数据，所以会用到天字段或小时字段分组统计）

– 首先将用户活跃度划分为每日活跃度与每时活跃度，并对比UV与PV的情况。

– 日活跃情况
– 我的sql(初步sql，没有参考价值)
select user_id,count(behavior_type) from userbehavior_dwd where behavior_type=’pv’ GROUP BY user_id
– 日活跃情况-实际sql
select behavior_date,count(DISTINCT user_id) as ‘UV’,count(behavior_type) as ‘PV’ from userbehavior_dwd
where behavior_type=’pv’ GROUP BY behavior_date

– 小时活跃情况
select HOUR(behavior_time) as ‘hour’,count(DISTINCT user_id) as ‘UV’,count(behavior_type) as ‘PV’ from userbehavior_dwd
where behavior_type=’pv’ GROUP BY hour

– 人均PV（日）
select behavior_date,count(DISTINCT user_id) as ‘UV’,count(behavior_type) as ‘PV’,
count(behavior_date)/count(DISTINCT user_id) as ‘MeanPVofUV’
from userbehavior_dwd
where behavior_type=’pv’ GROUP BY behavior_date
– 人均PV（小时）同上
– 分析价值：晚上20-22点的PV大幅上涨，是由于人均PV的上升产生的，说明晚上休息时间用户的活跃度相较白天有较大提升。

– 二、用户复购率
– （复购率对于平台来讲是非常重要的行为指标，复购率越高，平台的拉新成本，运营成本越低。）
– 逻辑分析：统计用户购买【相同商品/非相同商品】次数，用户复购率计算逻辑：复购人数/总购买人数
– 2.1、统计复购人数
– 2.2 统计总购买人数
– 2.3 计算复购率

select user_id,count(behavior_type) as ‘buy_num’ from userbehavior_dwd where behavior_type=’buy’ GROUP BY user_id
– 分析复购人员
select t.user_id,if(t.buy_num>1,1,0) as ‘复购人员’
from (select user_id,count(behavior_type) as ‘buy_num’ from userbehavior_dwd where behavior_type=’buy’ GROUP BY user_id ) t

– 统计复购人员数=复购人员总和
select sum(if(t.buy_num>1,1,0)) as ‘复购人数’
from (select user_id,count(behavior_type) as ‘buy_num’ from userbehavior_dwd where behavior_type=’buy’ GROUP BY user_id ) t

– 统计总购买人数
select count(DISTINCT user_id) as ‘总购买人数’ from userbehavior_dwd where behavior_type=’buy’

– 计算复购率=复购人员总和/总购买人数
select count(DISTINCT t.user_id) as ‘总购买人数’,sum(if(t.buy_num>1,1,0)) as ‘复购人数’,
CONCAT(ROUND(0/20*100,2),’%’) as ‘复购率’
from (select user_id,count(behavior_type) as ‘buy_num’ from userbehavior_dwd where behavior_type=’buy’ GROUP BY user_id ) t

– ▶▶购物环节转化率（AARRR分析模型）,主要分为浏览数据转化率、用户数据转化率
– 加购率=浏览数/加购数；购买率=购买数/加购数
– 浏览数据转化率
select pv_sum,cf_sum,buy_sum
,CONCAT(ROUND(cf_sum/pv_sum100,2),’%’) as cp
,CONCAT(ROUND(buy_sum/cf_sum100,2),’%’)as bc
from(
select
sum(IF(behavior_type=’pv’,1,0)) as pv_sum,
sum(IF(behavior_type=’cart’ or behavior_type=’fav’,1,0)) as cf_sum,
sum(IF(behavior_type=’buy’,1,0)) as buy_sum
from userbehavior_dwd
)t

– 用户数据转化率
select pv_sum,cf_sum,buy_sum
,CONCAT(ROUND(cf_sum/pv_sum100,2),’%’) as cp
,CONCAT(ROUND(buy_sum/cf_sum100,2),’%’)as bc
from(
select
COUNT(DISTINCT IF(behavior_type=’pv’,user_id,0)) as pv_sum,
COUNT(DISTINCT IF(behavior_type=’cart’ or behavior_type=’fav’,user_id,0)) as cf_sum,
COUNT(DISTINCT IF(behavior_type=’buy’,user_id,0)) as buy_sum
from userbehavior_dwd
)t

COUNT(behavior_type=’buy’) as buy_sum

– 用户价值指标（RFM分析模型），区分出：价值客户、发展客户、保持客户、挽留客户，一般挽留客户占比最高，进而是价值客户，发展客户，保持客户。
– 1、Recency-上次消费时间（间隔天数），求平均值
– 2、Frequency-消费频率，求平均值
– 3、R、F按价值打分规则&用户分组规则

– r_table、f_table –（TDM）
– r_table
CREATE VIEW R_table(user_id,recentbuy)
as
select u.user_id,
case
when recentbuy is not null then recentbuy
else 10
end as recentbuy
from userbehavior_dwd u left join
(
select user_id
,count(behavior_type) as recentbuy
from userbehavior_dwd
where behavior_type=’buy’
GROUP BY user_id
) t
on u.user_id=t.user_id
– GROUP BY u.user_id
ORDER BY u.user_id

– f_table
CREATE VIEW F_table(user_id,frequency)
as
select u.user_id,
case
when frequency is not null then frequency
else 10
end as frequency
from userbehavior_dwd u left join
(
select user_id
,MIN(DATEDIFF(‘2017-12-08’,behavior_date)) as frequency
from userbehavior_dwd
where behavior_type=’buy’
GROUP BY user_id
) t
on u.user_id=t.user_id
– GROUP BY u.user_id
ORDER BY u.user_id

– （ADM）
SELECT a.userID,r_score,f_score,
CASE
when r_score>2.35 and f_score>2.04 then ‘价值客户’
when r_score>2.35 and f_score<2.04 then ‘发展客户’
when r_score<2.35 and f_score>2.04 then ‘保持客户’
when r_score<2.35 and f_score<2.04 then ‘挽留客户’
END as userclass
FROM(
SELECT t.userID,r_score,f_score
FROM
(
SELECT userID,
CASE
when recentbuy=10 then 1
when recentbuy<10 and recentbuy>=5 then 2
when recentbuy<5 and recentbuy>=2 then 3
when recentbuy=1 then 4
END as r_score
FROM r_table
) t
LEFT JOIN
(
SELECT userID,
CASE
when frequency=0 then 1
when frequency<3 and frequency>=1 then 2
when frequency<7 and frequency>=3 then 3
when frequency>=7 then 4
END as f_score
FROM f_table
) f
ON t.userID=f.userID
) a
